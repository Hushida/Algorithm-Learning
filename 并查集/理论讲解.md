路径压缩（java实现）   
```
public QuickUnionUF {
    private int[] roots;
    
    public QuickUnionUF(int N) {
        roots = new int[N];
        for(int i = 0; i < N; i++) {
            roots[i] = i; //它的老大是他自己
        }
    }
    private int findRoot(int i) {
        int root = i;
        while(root != roots[root]) {
            root = roots[root];
        }
        while(i != roots[i]) {
            int temp = roots[i];
            roots[i] = root;
            i = temp;
        }
        return root;
    }
    public boolean connected(int p, int q) {
        return findRoot(p) == findRoot(q);
    }
    public void union(int p, int q) {
        int qroot = findRoot(q);
        int proot = findRoot(p);
        roots[proot] = qroot;
    }
}
```

伪代码
```
function MakeSet(x)
    x.parent := x //老大是他自己

function Find(x)
    if x.parent == x
        return x //判断老大是不是他自己
    else
        return Find(x.parent)//递归的查找下去
function Union(x, y)
    xRoot := Find(x)  //找到x的老大，找到y的老大
    yRoot := Find(y)
    xRoot.parent := yRoot
```
